  5     /*
  6      * Maps an objects values onto a new object, performing a shallow copy.
  7      * 
  8      * @param obj {Object} Object from which to map keys and values to a new
  9      * object. Assumes input is an associative array (Object) and not an 
 10      * Array.
 11      * @return {Object} A new object acting as a shallow copy clone of 
 12      * original.
 13      */    
 14     var shallowCopy = function(obj) {
 15         var out = {};
 16         var i;
 17         
 18         for (i in obj) {
 19             if (obj.hasOwnProperty(i)) {
 20                 out[i] = obj[i];
 21             }
 22         }
 23         
 24         return out;        
 25     };
 26     /*
 27      * Transposes an object's own properties, and value references, to a new 
 28      * object.
 29      * 
 30      * @param obj {Object} The object with which keys and values will be
 31      * transposed in the output. Assumes input is an associative array 
 32      * (Object) and not an Array.
 33      * @param [strict=false] {Boolean} If truthy, will throw an error if
 34      * object values do not map uniquely back to keys.
 35      * @return {Object} A new object with the keys and values inversed from
 36      * the original. Some object will always be created, even if obj passed
 37      * in is non-intelligent. Remember all keys will be of type String in
 38      * the output, because that's the way JavaScript rolls.
 39      */
 40     var transpose = function(obj, strict) {
 41         var out = {};
 42         var i;
 43         
 44         for (i in obj) {
 45             if (obj.hasOwnProperty(i)) {
 46                 if (strict && out.hasOwnProperty(i)) {
 47                     throw new Error("Tansposing non-distinct key:value pair :" + i);
 48                 }
 49                 out[obj[i]] = i;
 50             }
 51         }
 52         
 53         return out;
 54     };
 55     /*
 56      * Resets the value of any property of o1 to the value of o2, or sticks 
 57      * to the original value of o1 if the value does not exist on o2. Does 
 58      * not map new values from o1 to o2.
 59      * @param o1 {Object} The original object.
 60      * @param o2 {Object} The object containing reset values for the original.
 61      * @return {Object} A reference to o1, although since the change is in
 62      * place, the return value can be ignored.
 63      */
 64     var revalue = function(o1, o2) {
 65         var i;
 66         
 67         for (i in o1) {
 68             if (o1.hasOwnProperty(i) && o2.hasOwnProperty(i)) {
 69                 o1[i] = o2[i];
 70             }
 71         }
 72         return o1;
 73     };

130     /**
131      * Takes the provided "alphabet" of properties and, treating each value
132      * as a string, recodes each individual symbol of the alphabet heiroglyphs.
133      * @param alphabet {Object} The alphabet to rencode.
134      * @param symbolMap {Object} What to map the individual characters to.
135      * If the individual characters are not found in the symbolMap, they will
136      * be dropped through.
137      * @return {Object} The alphabet is returned, although as this is an
138      * in place change catching the return value is not entirely necessary.
139      */
140     var _encodeAlphabet = function(alphabet, symbolMap) {
141         var letter,
142             alphabetGlyph,
143             recodedGlyph,
144             i;
145             
146         for (letter in alphabet) {
147             if (alphabet.hasOwnProperty(letter)) {
148                 // Each glyph is assumed to be a string of symbols.
149                 alphabetGlyph = alphabet[letter].split("");
150                 recodedGlyph = [];
151                 for (i = 0; i < alphabetGlyph.length; i++) {
152                     recodedGlyph[i] = symbolMap[alphabetGlyph[i]] || alphabetGlyph[i];
153                 }
154                 // All symbols and glyphs are lower case. Correct here if
155                 // necessary. Case is set during the encoding process,
156                 // and case is not inherent to aural morse code.
157                 alphabet[letter] = recodedGlyph.join("").toLowerCase();
158             }
159         }
160         
161         return alphabet;
162     };

164     /**
165      * Encode a message into Morse Code.
166      * @param encodeFrom {string} To be translated into morse code.
167      * @returns {string} A translation of the text.
168      * @methodOf MorseCoder.prototype
169      */    
170     var encode = function(encodeFrom) {
171         // Array of strings, will be concated together for the output.
172         var out = [];
173         // Test to see if a letter is what we consider upper case.
174         var isUpper = /[A-Z]/;
175         // Reference to the encodings.
176         var alphabet = this.alphabet;
177         var gaps = this._gaps;
178         // Used to remove trailing gaps.
179         var gapRemover = new RegExp("("+gaps.letter+"|"+gaps.word+")*$");
180         // List of all space separated words.
181         var words = encodeFrom.split(/\s/);
182         // Loop counters and variables.
183         var word,
184             letter,
185             lowerCaseLetter,
186             encodedLetter,
187             i,
188             j;
189                     
190         // Iterate the entire string.
191         for (i = 0; i < words.length; i++) {
192             // Iterate over each word.
193             word = words[i];
194             // Only letters that can be mapped will be translated, if they can't
195             // they get dropped through.
196             for (j = 0; j < word.length; j++) {
197                 letter = word[j];
198                 // All encoding keys will be in lower case.
199                 lowerCaseLetter = letter.toLowerCase();
200                 if (lowerCaseLetter in alphabet) {
201                     // Known letters/symbols are encoded.
202                     encodedLetter = alphabet[lowerCaseLetter];
203                     // Note: Traditional encoding does not allow for upper
204                     // or lower case, but letter based encodings will.
205                     if (isUpper.test(letter)) {
206                         // We distinguish upper and lower case letters by
207                         // upper casing the visual output, something we can
208                         // do visually that is not possible aurally.
209                         encodedLetter = encodedLetter.toUpperCase();
210                     }
211                     out.push(encodedLetter);
212                 }
213                 else {
214                     // For unknown letters, we pass them through.
215                     out.push(letter);
216                 }
217                 
218                 // Are we at the end of the word?
219                 if (j != word.length-1) {
220                     // no we are not.
221                     out.push(gaps.letter);
222                 }
223                 else {
224                     // yes we are.
225                     out.push(gaps.word);
226                 }
227             }
228         }
229         // Remove the trailing "gaps" that might appear.
230         return out.join("").replace(gapRemover, "");
231     };

233     /**
234      * Encode a message from Morse Code.
235      * @param decodeFrom {string} To be translated from Morse Code.
236      * @returns {string} A decoding of the text.
237      * @methodOf MorseCoder.prototype
238      */
239     var decode = function(decodeFrom) {
240         // Array of strings, will be concated together for the output.
241         var out = [];
242         // Test to see if a letter is what we consider upper case.
243         var isUpper = /[A-Z]/;
244         // Reference to the encodings.
245         var alphabetDecoder = this.alphabetDecoder;
246         var gaps = this._gaps;
247         // List of all words determined by the output.
248         var words = decodeFrom.split(gaps.word);
249         // Loop counters.
250         var word,
251             letter,
252             lowerCaseLetter,
253             decodedLetter,
254             i,
255             j;
256         
257         for (i = 0; i < words.length; i++) {
258             // Word here is an array, not a string.
259             word = words[i].split(gaps.letter);
260             for (j = 0; j < word.length; j++) {
261                 // A letter here is not a single character but an encoding.
262                 letter = word[j];
263                 lowerCaseLetter = letter.toLowerCase();
264                 if (lowerCaseLetter in alphabetDecoder) {
265                     // Known encodings are decoded.
266                     decodedLetter = alphabetDecoder[lowerCaseLetter];
267                     // Note: Traditional encoding does not allow for upper
268                     // or lower case, but letter based encodings will.
269                     if (isUpper.test(letter)) {
270                         decodedLetter = decodedLetter.toUpperCase();
271                     }
272                     out.push(decodedLetter);
273                 }
274                 else {
275                     // Anything unknown passes right through.
276                     out.push(letter);
277                 }
278                 
279                 if (j == word.length-1) {
280                     // If we're at the end of the word, insert a space in
281                     // our buffer.
282                     out.push(" ");
283                 }
284             }
285         }
286         // Assume all words are already separated by spaces in our paradigm,
287         // and that trailing white space is not wanted.
288         return out.join("").trim();
289     };

291     /**
292      * @class Our Morse code encoder that can be overridden.
293      * @param [conf] {Object} Arguments as associative array.
294      * @param [conf.encodingOverrides] {Object} An associative array of the
295      * individual character overrides used to encode. The three characters
296      * recognized here are the dit ".", the dah "-", and the element gap " ".
297      * @param [conf.gapOverrides] {Object} An associative array of the
298      * gap overrides. The two gaps used are "letter" and "word".
299      *
300      * @constructor
301      * @name MorseCoder
302      */
303     exports.MorseCoder = function(cnf) {
304         var self = {};
305 
306         // Normalize.
307         cnf = cnf || {};
308 
309         // For diagnostic and purposes, we attach the defaults.
310         self._DEFAULT_MORSE_ENCODINGS = DEFAULT_MORSE_ENCODINGS;
311         self._DEFAULT_MORSE_CODE_ALPHABET = DEFAULT_MORSE_CODE_ALPHABET;
312         self._DEFAULT_MORSE_GAPS = DEFAULT_MORSE_GAPS;
313         
314         // Override the visual decodings of the letters.
315         self._encodings = shallowCopy(DEFAULT_MORSE_ENCODINGS);
316         if (cnf.encodingOverrides) {
317             revalue(self._encodings, cnf.encodingOverrides);
318         }
319         self._gaps = shallowCopy(DEFAULT_MORSE_GAPS);
320         if (cnf.gapOverrides) {
321             revalue(self._gaps, cnf.gapOverrides);
322         }
323         
324         // Per instance configurations.
325         self.alphabet = _encodeAlphabet( 
326             shallowCopy(DEFAULT_MORSE_CODE_ALPHABET),
327             self._encodings
328         );
329         self.alphabetDecoder = transpose(self.alphabet, true);
330         
331         // Attach our helpers.
332         self.encode = encode;
333         self.decode = decode;
334         // Attach our helpers.
335         
336         return self;
337     };
